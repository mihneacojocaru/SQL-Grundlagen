### in cazul in care avem mai mult de o conditie, putem folosi AND sau OR --> AND - ambele conditii trebuiesc indeplinite. OR - ori prima conditie, ori cealalta este suficienta pentru a satisface criteriul de sortare.

select * from masina where color in ('blue','puce','turquoise');
--> selecteaza toate elementele din masina unde culoarea este 'blue','puce' sau 'turquoise';

select * from masina where color not in ('blue','puce','turquoise');
--> selecteaza toate elementele in afara de cele care au culorile 'blue','puce' sau 'turquoise';

select * from masina where km between 20 and 60;
--> selecteaza toate elementele cu valoarea km cuprinsa intre 20 si 60;

select * from masina where marca like 'a%';
--> operatorul % alaturi de comanda LIKE alege toate valorile care incep cu caracterul preizat (in cazul nostru 'a'), urmat de orice '%';
operatorul % poate fi pus si inainte si dupa caracterul dorit: 
select * from masina where price like '%0,%'; pentru preturi gen: €0,5;
